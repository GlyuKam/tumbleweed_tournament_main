写给其他模组制作者：
欢迎大家在此模组上进行附加内容或额外游戏规则的修改！
为方便创作，下面将提供一些API和用法解读。


---------------------------------------------------------------------
--一些全局变量
---------------------------------------------------------------------
--游戏是否进行中-布尔型
TUNING.GZ_ON_GOING

--游戏是否已结束-布尔型
TUNING.GZ_GAME_OVER

--是否已执行过分组-布尔型
TUNING.GZ_HAS_GROUPED

--每组人数-自然数
TUNING.GZ_PLAYERS_EACH_GROUP

--是否允许投票-布尔型
TUNING.GZ_CAN_VOTE

--缩圈轮次-自然数
TUNING.GZ_COUNT_SUOQUAN

---------------------------------------------------------------------
--一些事件、注意这些事件在本模组内并没有监听、这些事件是给你开发别的模组的
---------------------------------------------------------------------
--缩圈事件
TheWorld:PushEvent('event_suoquan', TUNING.GZ_COUNT_SUOQUAN)

--游戏开始
TheWorld:PushEvent('event_GameStart')

--游戏结束-team_num：获胜队伍编号
TheWorld:PushEvent('event_GameOver', team_num)

---------------------------------------------------------------------
--说话
---------------------------------------------------------------------
DC_PlayerSay(player, msg)
--player：玩家实体
--msg：要说的话


---------------------------------------------------------------------
--切换投票开关
---------------------------------------------------------------------
gz_SwitchVote(bool)
--bool:开启或关闭-游戏准备的投票--与投票踢人没有任何关系


---------------------------------------------------------------------
--自动缩圈
---------------------------------------------------------------------
--执行一次标记缩圈点
gz_MarkShrink()

--执行一次标记终点点位
gz_MarkEnd()

--删除多余场地
gz_ClearSite()


---------------------------------------------------------------------
--观战
---------------------------------------------------------------------
gz_WatchGame(player)
--player：玩家实体


---------------------------------------------------------------------
--管理员出局
---------------------------------------------------------------------
gz_AdminOut()


---------------------------------------------------------------------
--尝试投票启动游戏
---------------------------------------------------------------------
gz_TryVoteStart()


---------------------------------------------------------------------
--投票启动游戏
---------------------------------------------------------------------
gz_VoteStart()


---------------------------------------------------------------------
--将玩家分组
---------------------------------------------------------------------
gz_GroupPlayers(each_num)
--each_num:数值：每个队伍分配人数


---------------------------------------------------------------------
--发放人数差补偿
---------------------------------------------------------------------
gz_GiveAid()


---------------------------------------------------------------------
--比赛开始
---------------------------------------------------------------------
gz_MatchStart(NoAuto, weed_num)
--NoAuto:bool型-不要自动游戏-也就是手动投放-通常填nil
--weed_num:每次投放数量-通常填nil


---------------------------------------------------------------------
--执行一次投放
---------------------------------------------------------------------
gz_Spawn(weed_num)
--weed_num:投放数量-通常填nil


---------------------------------------------------------------------
--开关自动投放
---------------------------------------------------------------------
在开启自动投放的状态下执行、则会关闭自动投放
在关闭自动投放的状态下执行、则会开启自动投放
gz_AutoSpawn(weed_num)
--weed_num:投放数量-通常填nil


---------------------------------------------------------------------
--设置投放间隔
---------------------------------------------------------------------
gz_SetSpawnPeriod(period, not_retask, weed_num)
--period:数值:投放间隔、单位秒
--not_retask:不要重启自动投放功能-通常填nil
--weed_num:投放数量-通常填nil


---------------------------------------------------------------------
--游戏结算
---------------------------------------------------------------------
gz_GameOver(team_num)
--team_num:胜出队伍编号


---------------------------------------------------------------------
--更改地面数字朝向
---------------------------------------------------------------------
gz_ReRotation()


---------------------------------------------------------------------
--增加风滚草产出
---------------------------------------------------------------------
gz_AddFgcLoot(tab)

--tab:传入一个表格
--chance:权重
--item:预制物名
--aggro:仇恨
--announce:公告
--pickfn:回调函数

--使用示例：
gz_AddFgcLoot(
    {
        --石头
        {chance = 25, item = 'flint'},
        --石头
        {chance = 25, item = 'flint'},
        --蜗牛壳碎片
        {chance = 2, item = 'slurper_pelt'},
        --啜食者皮
        {chance = 2, item = 'beardhair'},
        --牛
        {chance = .4, item = 'bunnyman', aggro = true},
        --兔人
        {chance = .4, item = 'tallbird', aggro = true},
        --高鸟
        {chance = .4, item = 'monkey', aggro = true},
        --猴子
        {chance = .4, item = 'rocky', aggro = true}
    }
)

---------------------------------------------------------------------
--API-移除风滚草产出
---------------------------------------------------------------------
--移除风滚草产出
gz_RemoveFgcLoot(loot)

--loot:预制物名

--使用示例：移除风滚草里的木头
gz_RemoveFgcLoot("log")
